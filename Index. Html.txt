import React, { useState, useMemo } from 'react';
import { 
  Heart, Droplets, Sparkles, Calendar as CalendarIcon, 
  Settings, ChevronLeft, ChevronRight, Plus, X, 
  Smile, Frown, Angry, CloudRain, Coffee
} from 'lucide-react';

// --- UTILITY FUNCTIONS ---
const normalizeDate = (d) => {
  const date = new Date(d);
  date.setHours(0, 0, 0, 0);
  return date;
};

const formatDateStr = (date) => {
  const d = new Date(date);
  const year = d.getFullYear();
  const month = String(d.getMonth() + 1).padStart(2, '0');
  const day = String(d.getDate()).padStart(2, '0');
  return `${year}-${month}-${day}`;
};

const addDays = (date, days) => {
  const result = new Date(date);
  result.setDate(result.getDate() + days);
  return result;
};

const getDaysDifference = (date1, date2) => {
  const diffTime = Math.abs(normalizeDate(date2) - normalizeDate(date1));
  return Math.ceil(diffTime / (1000 * 60 * 60 * 24));
};

export default function App() {
  // --- STATE MANAGEMENT ---
  const [isSetupComplete, setIsSetupComplete] = useState(false);
  const [cycleLength, setCycleLength] = useState(28);
  const [periodLength, setPeriodLength] = useState(5);
  const [lastPeriodDate, setLastPeriodDate] = useState(formatDateStr(new Date()));
  
  const [currentMonthDate, setCurrentMonthDate] = useState(normalizeDate(new Date()));
  const [selectedDate, setSelectedDate] = useState(null);
  const [showLogModal, setShowLogModal] = useState(false);
  const [showSettingsModal, setShowSettingsModal] = useState(false);
  
  // Logs memory structure: { 'YYYY-MM-DD': { mood: 'Happy', symptoms: ['Cramps'] } }
  const [logs, setLogs] = useState({});

  // --- CALCULATIONS ---
  const today = normalizeDate(new Date());
  
  // Predict upcoming dates based on last period
  const predictions = useMemo(() => {
    const lastP = normalizeDate(lastPeriodDate);
    const nextP = addDays(lastP, cycleLength);
    const nextOvulation = addDays(nextP, -14);
    const fertileStart = addDays(nextOvulation, -3);
    const fertileEnd = addDays(nextOvulation, 1);
    
    // Days until next period
    let daysUntilNext = getDaysDifference(today, nextP);
    if (today > nextP) {
       // If today is past the predicted next period, predict the one after
       daysUntilNext = cycleLength - (getDaysDifference(nextP, today) % cycleLength);
    }

    // Determine current phase
    let currentPhase = "Folikuler";
    let phaseDesc = "Waktu yang tepat untuk mencoba hal baru!";
    const daysSinceLastP = getDaysDifference(lastP, today);
    
    if (today >= lastP && today < addDays(lastP, periodLength)) {
      currentPhase = "Menstruasi";
      phaseDesc = "Waktunya istirahat dan manjakan dirimu 🌸";
    } else if (today >= fertileStart && today <= fertileEnd) {
      currentPhase = "Masa Subur";
      phaseDesc = "Energi sedang tinggi-tingginya! ✨";
    } else if (today > fertileEnd && today < nextP) {
      currentPhase = "Luteal";
      phaseDesc = "Mungkin terasa sedikit lelah, tetap terhidrasi ya 💧";
    }

    return { nextPeriod: nextP, fertileStart, fertileEnd, daysUntilNext, currentPhase, phaseDesc };
  }, [lastPeriodDate, cycleLength, periodLength, today]);

  // --- HANDLERS ---
  const handleSetupSubmit = (e) => {
    e.preventDefault();
    setIsSetupComplete(true);
  };

  const handleSaveLog = (dateStr, logData) => {
    setLogs(prev => ({
      ...prev,
      [dateStr]: logData
    }));
    setShowLogModal(false);
  };

  const prevMonth = () => {
    setCurrentMonthDate(addDays(currentMonthDate, -30)); // Approximate, fine for UI shift
  };

  const nextMonth = () => {
    setCurrentMonthDate(addDays(currentMonthDate, 30));
  };

  // --- CALENDAR GENERATION ---
  const renderCalendarDays = () => {
    const year = currentMonthDate.getFullYear();
    const month = currentMonthDate.getMonth();
    
    const firstDay = new Date(year, month, 1);
    const lastDay = new Date(year, month + 1, 0);
    const daysInMonth = lastDay.getDate();
    const startingDayOfWeek = firstDay.getDay(); // 0: Sun, 1: Mon...
    
    const days = [];
    
    // Empty cells for alignment
    for (let i = 0; i < startingDayOfWeek; i++) {
      days.push(<div key={`empty-${i}`} className="h-10 w-10"></div>);
    }
    
    // Day cells
    const lastP = normalizeDate(lastPeriodDate);

    for (let i = 1; i <= daysInMonth; i++) {
      const cellDate = new Date(year, month, i);
      const cellDateStr = formatDateStr(cellDate);
      const isToday = formatDateStr(cellDate) === formatDateStr(today);
      const hasLog = logs[cellDateStr] !== undefined;
      
      // Check if day is period or fertile (Predictive up to 3 cycles ahead)
      let isPeriod = false;
      let isFertile = false;
      
      for (let cycleOffset = 0; cycleOffset < 3; cycleOffset++) {
        const cycleStart = addDays(lastP, cycleOffset * cycleLength);
        const cycleEndPeriod = addDays(cycleStart, periodLength - 1);
        const ovulation = addDays(addDays(cycleStart, cycleLength), -14);
        const fStart = addDays(ovulation, -3);
        const fEnd = addDays(ovulation, 1);
        
        if (cellDate >= cycleStart && cellDate <= cycleEndPeriod) isPeriod = true;
        if (cellDate >= fStart && cellDate <= fEnd) isFertile = true;
      }

      let bgClass = "bg-white hover:bg-rose-50";
      let textClass = "text-slate-600";
      
      if (isPeriod) {
        bgClass = "bg-rose-400";
        textClass = "text-white font-semibold shadow-sm";
      } else if (isFertile) {
        bgClass = "bg-violet-200";
        textClass = "text-violet-800 font-medium";
      }

      days.push(
        <button
          key={i}
          onClick={() => {
            setSelectedDate(cellDateStr);
            setShowLogModal(true);
          }}
          className={`h-10 w-10 rounded-full flex flex-col items-center justify-center text-sm relative transition-all duration-200 ${bgClass} ${textClass} ${isToday ? 'ring-2 ring-offset-1 ring-rose-400 font-bold' : ''}`}
        >
          {i}
          {hasLog && (
            <span className="absolute bottom-1 w-1 h-1 rounded-full bg-slate-800"></span>
          )}
        </button>
      );
    }
    
    return days;
  };

  // --- SCREENS ---
  if (!isSetupComplete) {
    return (
      <div className="min-h-screen bg-rose-50 flex items-center justify-center p-4 font-sans text-slate-800">
        <div className="bg-white p-8 rounded-3xl shadow-xl shadow-rose-100 max-w-sm w-full border border-rose-100">
          <div className="text-center mb-6">
            <div className="bg-rose-100 w-16 h-16 rounded-full flex items-center justify-center mx-auto mb-4">
              <Sparkles className="text-rose-500 w-8 h-8" />
            </div>
            <h1 className="text-2xl font-bold text-rose-600">PeriodeKu 🌸</h1>
            <p className="text-sm text-slate-500 mt-2">Mari atur kalender estetikmu!</p>
          </div>
          
          <form onSubmit={handleSetupSubmit} className="space-y-4">
            <div>
              <label className="block text-sm font-medium text-slate-700 mb-1">Tanggal haid terakhir dimulai?</label>
              <input 
                type="date" 
                required
                value={lastPeriodDate}
                onChange={(e) => setLastPeriodDate(e.target.value)}
                className="w-full border-rose-200 rounded-xl shadow-sm focus:ring-rose-500 focus:border-rose-500 p-3 bg-rose-50 text-slate-700"
              />
            </div>
            <div className="flex gap-4">
              <div className="flex-1">
                <label className="block text-sm font-medium text-slate-700 mb-1">Siklus (Hari)</label>
                <input 
                  type="number" 
                  required
                  min="20" max="45"
                  value={cycleLength}
                  onChange={(e) => setCycleLength(parseInt(e.target.value))}
                  className="w-full border-rose-200 rounded-xl shadow-sm focus:ring-rose-500 focus:border-rose-500 p-3 bg-rose-50 text-slate-700 text-center"
                />
              </div>
              <div className="flex-1">
                <label className="block text-sm font-medium text-slate-700 mb-1">Lama Haid</label>
                <input 
                  type="number" 
                  required
                  min="2" max="10"
                  value={periodLength}
                  onChange={(e) => setPeriodLength(parseInt(e.target.value))}
                  className="w-full border-rose-200 rounded-xl shadow-sm focus:ring-rose-500 focus:border-rose-500 p-3 bg-rose-50 text-slate-700 text-center"
                />
              </div>
            </div>
            <button 
              type="submit" 
              className="w-full bg-rose-400 hover:bg-rose-500 text-white font-bold py-3 px-4 rounded-xl shadow-md shadow-rose-200 transition-colors mt-4"
            >
              Mulai Gunakan ✨
            </button>
          </form>
        </div>
      </div>
    );
  }

  return (
    <div className="min-h-screen bg-[#FFF5F8] text-slate-800 font-sans pb-20">
      {/* HEADER */}
      <header className="bg-white px-6 py-4 shadow-sm shadow-rose-100/50 sticky top-0 z-10 rounded-b-3xl flex justify-between items-center">
        <div className="flex items-center gap-2">
          <div className="bg-rose-100 p-2 rounded-full">
            <Heart className="w-5 h-5 text-rose-500 fill-rose-200" />
          </div>
          <h1 className="font-bold text-xl text-rose-600 tracking-tight">PeriodeKu</h1>
        </div>
        <button onClick={() => setShowSettingsModal(true)} className="p-2 text-slate-400 hover:bg-rose-50 rounded-full transition-colors">
          <Settings className="w-6 h-6" />
        </button>
      </header>

      <main className="max-w-md mx-auto p-4 space-y-6 mt-4">
        
        {/* DASHBOARD CARD */}
        <section className="bg-white p-6 rounded-3xl shadow-sm border border-rose-100 relative overflow-hidden">
          <div className="absolute top-0 right-0 w-32 h-32 bg-gradient-to-br from-rose-100 to-pink-50 rounded-bl-full -z-0 opacity-50"></div>
          
          <div className="relative z-10 flex flex-col items-center text-center">
            <h2 className="text-sm font-semibold text-rose-400 tracking-wide uppercase mb-1">Status Saat Ini</h2>
            <div className="text-4xl font-black text-slate-800 mb-2">
              {predictions.currentPhase === "Menstruasi" ? "Hari Haid" : `${predictions.daysUntilNext} Hari`}
            </div>
            <p className="text-slate-500 font-medium text-sm">
              {predictions.currentPhase === "Menstruasi" ? "Fase Menstruasi sedang berlangsung" : "Lagi menuju haid berikutnya"}
            </p>

            <div className="mt-6 w-full bg-rose-50 p-4 rounded-2xl border border-rose-100 flex items-start gap-3">
               <div className="bg-white p-2 rounded-full shadow-sm">
                 {predictions.currentPhase === "Menstruasi" ? <Droplets className="w-5 h-5 text-rose-500" /> : <Sparkles className="w-5 h-5 text-violet-400" />}
               </div>
               <div className="text-left">
                 <h3 className="font-bold text-slate-700">Fase {predictions.currentPhase}</h3>
                 <p className="text-xs text-slate-500 mt-1">{predictions.phaseDesc}</p>
               </div>
            </div>
            
            <button 
              onClick={() => {
                setLastPeriodDate(formatDateStr(today));
                setLogs(prev => ({...prev, [formatDateStr(today)]: { flow: 'Sedang', mood: 'Biasa' }}));
              }}
              className="mt-5 bg-rose-500 hover:bg-rose-600 text-white font-medium py-3 px-6 rounded-full shadow-lg shadow-rose-200/50 transition-all flex items-center gap-2"
            >
              <Droplets className="w-4 h-4 fill-white" />
              Haid Dimulai Hari Ini
            </button>
          </div>
        </section>

        {/* CALENDAR SECTION */}
        <section className="bg-white p-5 rounded-3xl shadow-sm border border-rose-100">
          <div className="flex justify-between items-center mb-4 px-2">
            <h2 className="font-bold text-lg text-slate-700 flex items-center gap-2">
              <CalendarIcon className="w-5 h-5 text-rose-400" />
              Kalender
            </h2>
            <div className="flex items-center gap-3">
              <button onClick={prevMonth} className="p-1 bg-rose-50 text-rose-600 rounded-full hover:bg-rose-100">
                <ChevronLeft className="w-5 h-5" />
              </button>
              <span className="font-medium text-sm text-slate-600 w-24 text-center">
                {currentMonthDate.toLocaleString('id-ID', { month: 'long', year: 'numeric' })}
              </span>
              <button onClick={nextMonth} className="p-1 bg-rose-50 text-rose-600 rounded-full hover:bg-rose-100">
                <ChevronRight className="w-5 h-5" />
              </button>
            </div>
          </div>

          {/* Legend */}
          <div className="flex justify-center gap-4 mb-4 text-xs font-medium text-slate-500">
            <div className="flex items-center gap-1"><span className="w-3 h-3 rounded-full bg-rose-400"></span> Haid</div>
            <div className="flex items-center gap-1"><span className="w-3 h-3 rounded-full bg-violet-200"></span> Subur</div>
          </div>

          <div className="grid grid-cols-7 gap-y-3 gap-x-1 justify-items-center mb-2">
            {['Min', 'Sen', 'Sel', 'Rab', 'Kam', 'Jum', 'Sab'].map(day => (
              <div key={day} className="text-xs font-bold text-slate-400 w-10 text-center">
                {day}
              </div>
            ))}
            {renderCalendarDays()}
          </div>
        </section>

      </main>

      {/* MODALS */}
      
      {/* Daily Log Modal */}
      {showLogModal && (
        <div className="fixed inset-0 bg-slate-900/40 backdrop-blur-sm flex items-end sm:items-center justify-center z-50 p-4">
          <div className="bg-white w-full max-w-md rounded-[2rem] p-6 shadow-2xl animate-in slide-in-from-bottom-10">
            <div className="flex justify-between items-center mb-6">
              <h3 className="font-bold text-xl text-slate-800">
                {new Date(selectedDate).toLocaleDateString('id-ID', { weekday: 'long', day: 'numeric', month: 'long' })}
              </h3>
              <button onClick={() => setShowLogModal(false)} className="p-2 bg-slate-100 rounded-full text-slate-500">
                <X className="w-5 h-5" />
              </button>
            </div>

            <div className="space-y-6">
              {/* Mood Selection */}
              <div>
                <label className="block text-sm font-semibold text-slate-700 mb-3">Suasana Hati (Mood)</label>
                <div className="flex gap-4">
                  {[
                    { icon: Smile, label: 'Senang', color: 'text-green-500', bg: 'bg-green-50' },
                    { icon: Frown, label: 'Sedih', color: 'text-blue-500', bg: 'bg-blue-50' },
                    { icon: Angry, label: 'Sensitif', color: 'text-red-500', bg: 'bg-red-50' }
                  ].map((m, i) => (
                    <button key={i} 
                      className={`flex-1 flex flex-col items-center p-3 rounded-2xl border-2 transition-all ${logs[selectedDate]?.mood === m.label ? `border-${m.color.split('-')[1]}-400 ${m.bg}` : 'border-slate-100 hover:bg-slate-50'}`}
                      onClick={() => handleSaveLog(selectedDate, { ...logs[selectedDate], mood: m.label })}
                    >
                      <m.icon className={`w-8 h-8 mb-2 ${m.color}`} />
                      <span className="text-xs font-medium text-slate-600">{m.label}</span>
                    </button>
                  ))}
                </div>
              </div>

              {/* Symptoms Selection */}
              <div>
                <label className="block t
